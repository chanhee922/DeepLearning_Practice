고객의 성별/연령별 분포

aa <- ggplot(data=top_buy, aes(reorder(CN_Name, sum, identity), sum))
aa + geom_col(fill = 'steelblue', col='black') + 
  labs(title = "상위 10개 최고 판매액 제품군", x = "제품군", y = "판매액") +

## 대분류별 판매건수 TOP 10

# 대대분류 판매 개수 top10
top_ct <- df_all %>%
  group_by(CN_Name) %>%
  summarise(sum=sum(PD_BUY_CT)) %>% 
  arrange(-sum) %>% 
  head(10)

bb <- ggplot(data=top_ct, aes(reorder(CN_Name, sum, identity), sum))
bb + geom_col(fill = 'steelblue',col='black') + 
  labs(title = "최다 판매 제품군 TOP10", x = "제품군", y = "판매 개수") +
  scale_fill_discrete(name = "제품군") +
  coord_flip()  + scale_color_economist() + theme_economist() + 
  theme(plot.title = element_text(face = "bold", hjust = 0.5, size = 20, color = "black")) +
  theme(axis.title = element_text(size = 12, color = "black")) +
  scale_y_continuous(labels = scales::comma)

### 인사이트 - 매출액이 패션에 크게 치중되어 있다. - 전자제품의 경우, 매출액은 높지만 판매 건수는 상대적으로 낮다 - 유아용품의 경우, 매출액은 낮지만 판매 건수는 상대적으로 많다

연령대별 전체 고객 최고 판매액 제품군

temp <- df_all %>%
  group_by(CLNT_AGE, CN_Name) %>%
  summarise(판매액=sum(PD_BUY_TOT)) %>% 
  arrange(CLNT_AGE, -판매액) %>% 
  mutate(rank=row_number()) %>% 
  filter(rank %in% c(1,2,3))

ggplot(temp, aes(x = CLNT_AGE, y = 판매액, fill = CN_Name)) +
  geom_bar(stat = "identity",col='black') +
  labs(title = "전체 고객 최고 판매액 제품군", x = "연령대", y = "판매액") +
  scale_fill_discrete(name = "제품군") + scale_color_economist() + theme_economist() + 
  theme(plot.title = element_text(face = "bold", hjust = 0.5, size = 20, color = "black")) +
  theme(axis.title = element_text(size = 16, color = "black")) +
  scale_y_continuous(labels = scales::comma)

인사이트

  • 전체 및 남성 30대 고객과 비교했을 때, 30대 여성의 유아용품 구매가격이 눈에 띈다

* 전체적으로 패션 카테고리의 판매액이 도드라진다. 그렇다면, 판매액이 높은 고객들을 VIP로 구분해 클러스터링을 진행한 결과는 어떻게 될까?

데이터 설명

vip_train은 구매 데이터에서 구매액 기준 박스플롯에서 최댓값 이상의 고객을 VIP로 규정해 담은 데이터 프레임이다. vip_train$class에는 군집화 한 결과가 1과 2로 구분되어져있다:

데이터 정제

레이더 차트 함수 raiderchart{fmsb}에 맞는 데이터 프레임 구조를 만들기 위해 vip_train을 다음과 같이 정제했다:

radar <- vip_train %>%
  group_by(class) %>% 
  summarise(
      전자제품 = sum(C1_CT),
      리빙 = sum(C2_CT), 
      식품 = sum(C3_CT),
      스포츠 = sum(C4_CT),
      패션 = sum(C5_CT),
      뷰티 = sum(C6_CT),
      유아용품 = sum(C7_CT),
      취미용품 = sum(C8_CT),
      사무용품 = sum(C9_CT),
      상품권 = sum(C10_CT) )
radar = as.data.frame(radar)
radar <- radar[-1]

결과

정제한 데이터 프레임을 단순히 raiderchart()에 넣었을때 다음과 같은 차트가 나온다:

radarchart(radarC)

보기 쉽게 여러 인자를 추가할 경우 다음과 같은 결과가 나온다:

# 색 설정
colors_border=c( rgb(0.2,0.5,0.5,0.9), rgb(0.8,0.2,0.5,0.9) , rgb(0.7,0.5,0.1,0.9) )
colors_in=c( rgb(0.2,0.5,0.5,0.4), rgb(0.8,0.2,0.5,0.4) , rgb(0.7,0.5,0.1,0.4) )

# 레이더 차트 생성
radarchart( radarC  , axistype=1 , 
                 #custom polygon
                   pcol=colors_border , pfcol=colors_in , plwd=4 , plty=1,
                 #custom the grid
                   cglcol="grey", cglty=1, axislabcol="grey", caxislabels=seq(0,80000,20000), cglwd=0.8,
                 #custom labels
                   vlcex=0.8 
               )
               # caxislabels=seq(0,80000,20000) : 중앙부터 8만까지 2만의 단위로 선을 긋는다

# 범주(legend) 생성
 legend(x='topright', y=1, legend = rownames(radarC[c(3,4),]), bty = "n", pch=20 ,
                 col=colors_in , text.col = "grey", cex=1.2, pt.cex=3)

인사이트

  • vip_train에서 클러스터링 된 그룹 cluster 1은 패션에 치우친 구매 패턴을 보인다.
  • cluster 2는 상품권, 사무용품, 취미용품을 제외한 전 분야에서 고르게 구매를 하는 패턴을 보인다.
